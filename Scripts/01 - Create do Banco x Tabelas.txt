/* Tabela de Estados */
CREATE TABLE estados(
	est_id	  INT PRIMARY KEY IDENTITY,
	est_nome  VARCHAR(40) NOT NULL,
	est_sigla CHAR(2) NOT NULL
)
GO

/* Tabela de Cidades */
CREATE TABLE cidades(
	cid_id     INT PRIMARY KEY IDENTITY,
	cid_nome   VARCHAR(100) NOT NULL,
	cid_est_fk INT NOT NULL
)
GO
